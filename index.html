sudo nano /etc/nginx/sites-available/domain.com
# Nginx Configuration for domain.com
server {
listen
80
server_name
domain.com www.domain.com

# ===== Frontend (Next.js) =====
location
/ {
proxy_pass
http://localhost:3000;
proxy_http_version
1.1;

Upgrade $http_upgrade;
proxy_set_header
Connection 'upgrade';

Host $host;
proxy_cache_bypass
$http_upgrade;

location
/api/ {
proxy_pass
http://localhost:5000/;

Host $host;

/images/ {

off;

Auto Deployment Setup

Deployment Script
Create automated deployment script
1
nano deploy.sh
deploy.sh
#!/bin/bash

# Change to project directory
cd /root/test-ci-cd || exit

# Update from GitHub
git fetch --all
git reset --hard origin/main
git pull origin main

# Install dependencies
npm install

# Restart or start PM2 process
pm2 restart test-ci-cd || pm2 start npm --name "test-ci-cd" -- start

pm2 save
2
chmod +x /root/test-ci-cd/deploy.sh
3
./deploy.sh
4
pm2 list

Webhook Setup

Webhook Server
Create GitHub webhook receiver
1
nano webhook.js
webhook.js
const express = require("express");
const { exec } = require("child_process");

const app = express();
app.use(express.json());

app.post("/deploy", (req, res) => {
exec("sh /root/test-ci-cd/deploy.sh", (err, stdout, stderr) => {
if (err) {
console.error(stderr);
return res.status(500).send(stderr);
}
console.log(stdout);
res.send("Deployed successfully");
});
});

app.listen(5001, () => console.log("Webhook running on port 5001"));
2
pm2 start /root/test-ci-cd/webhook.js --name deploy-webhook
3
pm2 save
